Buttons
	*plantSeed
		name:Plant a Tree
		desc:Click here to plant a tree!
		tag:planting
		on click:
			if (seeds > 0)
				yield 1 planted
				yield 1 tree
				lose 1 seed
			end
			if (seeds = 0) toast Not enough seeds. Gather more!
		end
		icon:icons[0,1]
		class:noBackground
		//no text
		//class:bigButton
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

	*gatherSeed
		name:Gather Seeds
		desc:Click here to gather some seeds!
		tag:gathering
		on click:
			if (gatherCooldown = 0)
				yield trees*random(2,4) seeds
				gatherCooldown = 20
				hide gatherSeed
			end
		end
		icon:icons[0,0]
		class:noBackground
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red
		hidden
		on tick:
			if (have upgradeGathering and gatherCooldown=0) show gatherSeed
		end

	*cutTree
		name:Cut a Tree
		desc:Click here to cut down a tree and gain 1 wood.
		tag:cutting
		hidden
		on tick:
			if (trees > 100) show cutTree
		end
		on click:
			if (trees > 0)
				yield 1 wood
				lose 1 tree
			end
		end
		icon:icons[1,1]
		class:noBackground
		//no text
		//class:bigButton
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

Buildings
	*gatherer
		name:Gatherer
		desc: gather 2-3 seeds per second.
		icon:icons[0,0]
		class:noBackground
		on tick:
			yield random(2,3) seeds
		end
		cost:5 trees

	*forester
		name:Forester
		desc: The Forester plants 1 tree per second at a cost of 1 money. Enough seeds must be available.
		icon:icons[1,0]
		class:noBackground
		on tick:
			if (seeds>0)
				yield 1 trees
				yield 1 planted
				lose 1 seed
				lose 1 money
			end
		end
		cost:5 trees

	*woodcutter
		name:Woodcutter
		desc: The Woodcutter cuts 1 tree every 5 seconds to gain 1 wood at a cost of 20 money.
		icon:icons[1,1]
		class:noBackground
		on tick:
			if (trees>0)
				yield 0.2 wood
				lose 0.2 tree
				lose 4 money
			end
		end
		cost:100 wood
		req:1000 trees
